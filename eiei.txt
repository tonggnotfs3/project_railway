SELECT complaintform.complaint_name,complaintform.personal_id,complaintform.address,province.PROVINCE_NAME,amphur.AMPHUR_NAME,districts.DISTRICT_NAME,complaintform.tel,complaintform.email, complainttype.complaint_code,service.service_code,complaintform.description FROM complaintform INNER JOIN (districts,province,amphur,complainttype,service,tbstatus) ON (districts.DISTRICT_ID = complaintform.districts_id,province.PROVINCE_ID = complaintform.province_id,amphur.AMPHUR_ID = complaintform.amphur_id,complainttype.complainttype_id = complaintform.complainttype_id,service.complaint_id = complaintform.service_id,tbstatus.status_id = complaintform.status_id)
